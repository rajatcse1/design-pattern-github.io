/*
  Create DataOutputStream from FileOutputStream
  This Java example shows how to create DataOutputStream from FileOutputStream
  object.
*/
 
import java.io.DataOutputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
 
public class CreateDataOutputStream {
 
  public static void main(String[] args) {
   
    String strFilePath = "C://FileIO//demo.txt";
   
    try
    {
      //create FileOutputStream object
      FileOutputStream fos = new FileOutputStream(strFilePath);
     
      /*
       * To create DataOutputStream object from FileOutputStream use,
       * DataOutputStream(OutputStream os) constructor.
       *
       */
     
       DataOutputStream dos = new DataOutputStream(fos);
    }
    catch(FileNotFoundException fe)
    {
      System.out.println("FileNotFoundException : " + fe);
    }
 
  }
}
